void change(int i,int j)
{
	int a[20]={0};//为数组赋予初值零，即使出错时输出也不会杂乱无章
	int k=0;
	printf("十进制数%d转换为%d进制后为：",i,j);
	while(i)
	{
		a[k] = i % j;//“==”是判断数值上是否相等，而此处应用赋值运算符“=”，将表达式的值赋给a[k]
		k++;         //最终k指向了a中有效元素的下一元素，所以下面的for循环应先使k-1
		i = i/j;
	}

	for(k=k-1;k>=0;k--)
	{
		printf("%d",a[k]);
	}
}
